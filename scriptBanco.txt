CREATE TABLE `prova_dev_carlos`.`tb_usuarios` ( `ID_USUARIO` INT NOT NULL AUTO_INCREMENT , `NOME_COMPLETO` VARCHAR(200) NOT NULL , `CEP` INT(8) NOT NULL , `EMAIL` VARCHAR(100) NOT NULL , `SENHA` VARCHAR(100) NOT NULL , `USUARIO` VARCHAR(40) NOT NULL , `CRIADO_EM` DATE NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`ID_USUARIO`)) ENGINE = InnoDB;
